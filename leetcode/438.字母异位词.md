## 我的

```c
class Solution {
public:
    vector<int> findAnagrams(string s, string p) {
        int i=0,j=0;
        vector<int> ans;
        unordered_multiset<char> uset;
        while(i<s.size()&&j<s.size()){
            if(p.find(s[j])!=std::string::npos&&uset.find(s[j])==uset.end()){
                uset.insert(s[j]);
                j++;
            }
            else{
                uset.clear();
                j=i+1;
                i=j;
            }
            if(j-i==p.size()){
                uset.clear();
                ans.push_back(i);
                j=i+1;
                i=j;
            }
        }
        return ans;
    }
};
```

使用这种会导致无法对p="baa"这种包含重复字符的进行判断

在对数组名称使用 `==` 运算符时，运算符会比较数组的开始内存地址，而不是数组的内容。

```c
class Solution {
public:
    vector<int> findAnagrams(string s, string p) {
        vector<int> scnt(26,0), pcnt(26),ans;
        for(int i=0;i<p.size();i++) pcnt[p[i]-'a']++;
        for(int i=0,j=0;i<s.size();i++){
            while(j<s.size()){
                scnt[s[j]-'a']++;
                j++;
                if(j-i==p.size()){
                    if(scnt==pcnt){
                        ans.push_back(i);
                    }
                    scnt[s[i]-'a']--;  // !!! 一个个把i收回
                    break;
                }  
            }
            
        }
        return ans;
    }
};
```
