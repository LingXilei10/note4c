class Solution {
public:
    int fib(int n) {
        if(n==0||n==1) return n;
        int f_i_1=0,f_i_2=1;
        for(int i=2;i<=n;i++){
            int f_i = f_i_1+f_i_2;   // 1 2 3
            f_i_1 = f_i_2;  // 2
            f_i_2 = f_i;  // 3
        }
        return f_i_2;
    }
};
