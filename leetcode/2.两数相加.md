/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        ListNode* dummynode = new ListNode(0);
        ListNode* cur = dummynode;
        int flag=0;
        int he = 0;
        while(l1!=nullptr&&l2!=nullptr){
            
            if(l1->val+l2->val+flag<10) {
                he = l1->val+l2->val+flag;
                flag=0;
            }
            else {
                he = (l1->val+l2->val+flag)%10;
                flag = 1;
            }
            ListNode *newnode = new ListNode(he);
            cur->next = newnode;
            cur = cur->next;
            l1 = l1->next;
            l2 = l2->next;
        }
        if(l1==nullptr){
            while(l2!=nullptr){
                if(l2->val+flag<10) {
                    he = l2->val+flag;
                    flag=0;
                }
                else {
                    he = (l2->val+flag)%10;
                    flag = 1;
                }
                ListNode *newnode = new ListNode(he);
                cur->next = newnode;
                cur = cur->next;
                l2 = l2->next;
            }
        }
        else if (l2==nullptr){
            while(l1!=nullptr){
                if(l1->val+flag<10) {
                    he = l1->val+flag;
                    flag=0;
                }
                else {
                    he = (l1->val+flag)%10;
                    flag = 1;
                }
                ListNode *newnode = new ListNode(he);
                cur->next = newnode;
                cur = cur->next;
                l1 = l1->next;
            }
        }
        if(flag){
            ListNode *newnode = new ListNode(flag);
            cur->next = newnode;
            cur = cur->next;
        }
        return dummynode->next;
    }
};
