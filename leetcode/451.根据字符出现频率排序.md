#include <vector>
#include <string>
#include <unordered_map>
#include <queue>
#include <functional>

class Solution {
public:
    std::string frequencySort(std::string s) {
        std::vector<char> chars(s.begin(), s.end());
        // s ä¸­çš„å­—ç¬¦ -> è¯¥å­—ç¬¦å‡ºç°çš„é¢‘ç‡
        std::unordered_map<char, int> charToFreq;
        for (char ch : chars) {
            charToFreq[ch]++;
        }

        auto comp = [](const auto& a, const auto& b) {
            // é˜Ÿåˆ—æŒ‰ç…§é”®å€¼å¯¹ä¸­çš„å€¼ï¼ˆå­—ç¬¦å‡ºç°é¢‘ç‡ï¼‰ä»å¤§åˆ°å°æ’åº
            return a.second < b.second;
        };
        std::priority_queue<std::pair<char, int>, std::vector<std::pair<char, int>>, decltype(comp)> pq(comp);

        // æŒ‰ç…§å­—ç¬¦é¢‘ç‡æ’åº
        for (const auto& entry : charToFreq) {
            pq.emplace(entry.first, entry.second);
        }

        std::string sb;
        while (!pq.empty()) {
            // æŠŠé¢‘ç‡æœ€é«˜çš„å­—ç¬¦æ’åœ¨å‰é¢
            auto entry = pq.top();
            pq.pop();
            sb.append(entry.second, entry.first);
        }

        return sb;
    }
};
